== bool 类型

[source, java]
----
bool a = 23 == 45; // 判断 23 和45 是否相等, 将结果赋给一个布尔类型的变量
Console.WriteLine(a); //False
----



---


== 逻辑运算(&&是, ||或, !非)

[source, java]
----
bool a = (3 < 4) && (9 < 6); // && 是前后两个都为true时, 才最终为ture.
Console.WriteLine(a); //False

bool a = (3 < 4) || (9 < 6);
Console.WriteLine(a); //True

bool a = !(3 < 5); //3<5为ture, 但前面加个!, 就是取非了, 就变成了!ture=false
Console.WriteLine(a); //False
----
